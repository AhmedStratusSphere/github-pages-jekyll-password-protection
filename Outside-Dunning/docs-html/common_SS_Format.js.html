/**
 * Wrapper module for N/format
 *
 * @NApiVersion     2.1
 * @NModuleScope    SameAccount
 *
 * @fileoverview    Wrapper module for N/format
 * @module          SS_Format
 * @public
 *
 * @requires        N/format
 *
 * @return {Object}  SS_Format
 * @return {Function} SS_Format.clean
 * @return {Function} SS_Format.dateToString
 * @return {Function} SS_Format.format
 * @return {Function} SS_Format.isEmpty
 * @return {Function} SS_Format.normalizeDate
 * @return {Function} SS_Format.parse
 *
 *
 */

define(
    [
        'N/format',
    ],
    (
        NS_Format
    ) => {

        /**
         * @name MODULE
         * @description Module name for logging
         * @type {string}
         *
         * @constant
         *
         * @example MODULE = 'outside.Format'
         */
        const MODULE = 'outside.Format';

        /**
         * @function cleanObject
         * @param {Object} param
         * @description Clean an object by removing any empty properties
         * @returns {Object}
         * @example cleanObject({ key: 'value', empty: '' });
         */
        const cleanObject = (param) => {
            if (isEmpty(param)) { return null; }

            for (const [key, value] of Object.entries(param)) {
                if (isEmpty(value)) {
                    delete param[key];
                }
            }
            if (isEmpty(param)) { return null; }

            return param;
        };

        /**
         * @function cleanString
         * @param {string} value
         * @param {string} delimiter
         * @description Clean a string by removing spaces, slashes, pipes, and underscores
         * @returns {string}
         * @example cleanString('This is a string');
         * @example cleanString('This is a string', '_');
         * @example cleanString('This is a string', '-');
         *
         */
        const cleanString = (value, delimiter) => {
            if (!value) { return ''; }

            let delim = delimiter || '';
            let output = value?.toLowerCase();
            output = output.replaceAll(' ', delim);
            output = output.replaceAll('/', delim);
            output = output.replaceAll('|', delim);
            output = output.replaceAll('_', delim);

            return output;
        };

        /**
         * @function dateToString
         * @param {Date} date
         * @description Convert a date to a string
         * @returns {string}
         * @example dateToString(new Date());
         * @example dateToString(new Date('2021-01-01'));
         */
        const dateToString = (date) => {
            if (!date) { return '' };
            return NS_Format.format({ type: NS_Format.Type.DATE, value: date });
        };

        /**
         * @function formatCurrency
         * @param {Object} options
         * @param {string} options.symbol
         * @param {number} options.value
         * @param {boolean} options.hideSymbol
         * @description Format a currency value
         * @returns {string}
         * @example formatCurrency({ value: 1000 });
         * @example formatCurrency({ value: 1000, symbol: 'Â£' });...
         */
        const formatCurrency = (options) => {
            let symbol = options.hideSymbol === true ? '' : (options.symbol || '$');
            let inputValue = options.value || 0;
            if (inputValue == 0) {
                return `${symbol}0.00`;
            }
            else {
                return `${symbol}${NS_Format.format({ type: NS_Format.Type.CURRENCY, value: inputValue })}`;
            }
        };

        /**
         * @function formatDate
         * @param {Object} options
         * @param {Date} options.value
         * @description Format a date value
         * @returns {string}
         * @example formatDate({ value: new Date() });
         * @example formatDate({ value: new Date('2021-01-01') });
         * @example formatDate({ value: null });
         */
        const formatDate = (options) => {
            return options.value ? NS_Format.format({ type: NS_Format.Type.DATE, value: options.value }) : '';
        };

        /**
         * @function formatFloat
         * @param {Object} options
         * @param {number} options.value
         * @description Format a floating point number
         * @returns {string}
         */
        const formatFloat = (options) => {
            return options.value ? NS_Format.format({ type: NS_Format.Type.FLOAT, value: options.value }) : '';
        };

        /**
         * @function formatInteger
         * @param {Object} options
         * @param {number} options.value
         * @description Format an integer
         * @returns {string}
         * @example formatInteger({ value: 1000 });
         * @example formatInteger({ value: 1000.00 });
         */
        const formatInteger = (options) => {
            return options.value ? NS_Format.format({ type: NS_Format.Type.INTEGER, value: options.value }) : '';
        };

        return {
            get Type() {
                return NS_Format.Type;
            },
            clean(param) {
                let output = null;
                let type = Object.prototype.toString.call(param);
                switch (type) {
                    case '[object Object]': {
                        output = cleanObject(param);
                        break;
                    }
                    case '[object String]': {
                        output = cleanString(param);
                        break;
                    }
                }

                return output;
            },
            dateToString,
            format(options) {
                switch (options.type) {
                    case this.Type.CURRENCY: {
                        return formatCurrency(options);
                    }
                    case this.Type.DATE: {
                        return formatDate(options);
                    }
                    case this.Type.FLOAT: {
                        return formatFloat(options);
                    }
                    case this.Type.INTEGER: {
                        return formatInteger(options);
                    }
                    default: {
                        return options.value;
                    }
                }
            },
            isEmpty(value) {
                let output = false;
                let type = Object.prototype.toString.call(value);
                switch (type) {
                    case '[object Object]': {
                        if (Object.keys.length <= 0) {
                            output = true;
                        }
                        break;
                    }
                    case '[object String]': {
                        if (value == '' || value == null || value == undefined) {
                            output = true;
                        }
                        else if (value.trim() == '') {
                            output = true;
                        }
                        break;
                    }
                }

                return output;
            },
            normalizeDate(options) {
                let date = NS_Format.parse({ type: this.Type.DATE, value: options.value });
                let output = dateToString(date);
                return output;
            },
            parse(options) {
                return NS_Format.parse(options)
            }
        }
    }
);
