/**
 * @NApiVersion     2.1
 * @NScriptType     Suitelet
 *
 * @module          SS_SL_DataTables
 * @public
 * @description   This file contains the code for the Suitelet that will render the DataTables Suitelet.
 *
 * @requires SS_DataTableSuitelet
 *
 */

define(
    [
        '../common/SS_DataTableSuitelet'
    ],
    (
        DataTableSuitelet
    ) => {
        /**
         * @name MODULE
         * @description Constant variable to store the module name.
         * @type {string}
         */
        const MODULE = 'outside.SL|DataTables';

        /**
         * @function onRequest
         * @description Function that will be called when the Suitelet is accessed.
         * @param {Object} options
         * @param {ServerRequest} options.request
         * @param {ServerResponse} options.response
         * @returns {void}
         */
        const onRequest = (options) => {
            const title = `${MODULE}.Request`;
            log.debug( 'option', JSON.stringify(options));
            DataTableSuitelet.render(options);
        };

        return { onRequest };
    }
);
