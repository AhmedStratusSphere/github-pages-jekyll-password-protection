/**
 * Wrapper module for N/query
 *
 * @NApiVersion     2.1
 * @NModuleScope    SameAccount
 *
 * @module          SS_Query
 * @public
 * @exports SS_Query
 * @requires N/query
 *
 * @return {Object} - Module functions
 */

define(
    [
        'N/query'
    ],
    (
        NS_Query
    ) => {

        /**
         *  SQL query to get all active currencies from the Currency table
         * @name   CURRENCIES
         * @type {String}
         * @constant
         *
         */
        const CURRENCIES = `SELECT
            Symbol,
            Name,
            ExchangeRate,
            DisplaySymbol,
            SymbolPlacement,
            CurrencyPrecision,
            IsBaseCurrency
        FROM
            Currency
        WHERE
            ( IsInactive = 'F' )
        ORDER BY
            Symbol`;

        /**
         * SQL query to get all values from a custom list
         * @name CUSTOM_LIST_VALUES
         * @constant
         * @type {string}
         *
         *
         */
        const CUSTOM_LIST_VALUES = `
        SELECT
            Name, 
            ID
        FROM
            {CUSTOM_LIST_ID}
        WHERE
            IsInactive = 'F'
        ORDER BY
            Name
        `;

        /**
         * SQL query to get all active custom lists
         * @name CUSTOM_LISTS
         * @type {string}
         * @constant
         */
        const CUSTOM_LISTS = `SELECT
            Name,
            Description,
            ScriptID,
            BUILTIN.DF( Owner ) AS Owner,
            IsOrdered
        FROM 
            CustomList
        WHERE
            ( IsInactive = 'F' )
        ORDER BY
            Name`;

        /**
         * SQL query to get all countries and states
         * @name STATE_COUNTRIES
         * @constant
         * @type {string}
         */
        const STATE_COUNTRIES = `SELECT
            Country.ID AS CountryID,
            Country.Name AS CountryName,
            Country.Edition,
            Country.Nationality,
            State.ID AS StateID,
            State.ShortName AS StateShortName,
            State.FullName AS StateFullName
        FROM
            Country
            LEFT OUTER JOIN State ON
                ( State.Country = Country.ID )
        ORDER BY
            CountryName,
            StateShortName`;

        return {
            get Currencies() {
                return this.run(CURRENCIES);
            },
            get CustomLists() {
                return this.run(CUSTOM_LISTS);
            },
            get StatesWithCountries() {
                return this.run(STATE_COUNTRIES);
            },

            getCustomListValues(id) {
                return this.run(CUSTOM_LIST_VALUES.replace('{CUSTOM_LIST_ID}', id));
            },
            run(query) {
                let queryObject = NS_Query.runSuiteQL({ query });
                return queryObject.asMappedResults();
            },
            runWithParams(options) {
                let { query, params } = options;
                let queryObject = NS_Query.runSuiteQL({ query, params });
                return queryObject.asMappedResults();
            }
        };
    }
);
