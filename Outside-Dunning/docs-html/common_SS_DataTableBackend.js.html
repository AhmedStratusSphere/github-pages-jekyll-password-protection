/**
 * @NApiVersion         2.1
 * @NModuleScope        SameAccount
 *
 * @Module           SS_DataTableBackend
 *
 *
 * @description        SS_DataTableBackend module for handling DataTable Suitelet backend processing
 *
 * @requires         SS_Constants
 * @requires         SS_File
 * @requires         SS_Script
 * @requires         SS_Search
 * @requires         SS_String
 * @requires         SS_UI
 * @requires         SS_Url
 * @requires         SS_DunningStatements
 * @requires         SS_DunningLetters
 */

define(
    [
        './SS_Constants',
        './SS_File',
        './SS_Script',
        './SS_Search',
        './SS_String',
        './SS_UI',
        './SS_Url',

        './SS_DunningStatements',
       './SS_DunningLetters'
    ],
    (
        SS_Constants,
        SS_File,
        SS_Script,
        SS_Search,
        SS_String,
        SS_UI,
        SS_Url,

        SS_DunningStatements,
        SS_DunningLetters,

    ) => {

        /**
         * @name MODULE
         * @type {string}
         * @description constant Module name for logging
         *@constant
         * @example
         * MODULE = `outside|DataTableBackend`;
         * log.debug({ title: MODULE, details: 'message' });
         */
        const MODULE = `SS|DataTableBackend`;

        /**
         * @name TYPES
         * @description constant declaration that take DataTable suitelet types create on the custom record for processing from SS_Constants module
         * @see {@link https://5678674-sb2.app.netsuite.com/app/common/search/ubersearchresults.nl?quicksearch=T&searchtype=Uber&frame=be&Uber_NAMEtype=KEYWORDSTARTSWITH&Uber_NAME=SS%20%7C%20Data%20Table%20Suitelet%20Types%09}
         * @type {Object}
         * @constant
         * @example
         * const TYPES = SS_Constants.CustomLists.DataTableSuiteletTypes;
         */
        const TYPES = SS_Constants.CustomLists.DataTableSuiteletTypes;

        /**
         * @name HANDLERS
         * @description constant object that contains the handlers for the DataTable Suitelet send actions from SS_DunningStatements and SS_DunningLetters module
         *
         * @constant
         * @example
         * const HANDLERS = {
         *    'dunning_statement': SS_DunningStatements.send,
         *    'dunning_letter': SS_DunningLetters.send
         *    }
         * @type {Object}
         *
         */
        const HANDLERS = {
            'dunning_statement': SS_DunningStatements.send,
            'dunning_letter': SS_DunningLetters.send
        };


        /**
         * @function buildSearchFilters
         * @description function that build the search filters for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.parameters - The parameters for the DataTable Suitelet
         * @returns {Array} filters - The filters for the DataTable Suitelet
         * @example
         *         const buildSearchFilters = (options) => {
         *             const TITLE = `${MODULE}.BuildSearchFilters`;
         *             let { parameters } = options;
         *             let filters = [];
         *
         *             switch (parameters.type) {
         *                 case TYPES.DUNNING_STATEMENTS: {
         *                     filters = SS_DunningStatements.buildSearchFilters(parameters);
         *                     break;
         *                 }
         *             }
         *
         *             return filters;
         *         };
         */
        const buildSearchFilters = (options) => {
            const TITLE = `${MODULE}.BuildSearchFilters`;
            let { parameters } = options;
            let filters = [];

            switch (parameters.type) {
                case TYPES.DUNNING_STATEMENTS: {
                    filters = SS_DunningStatements.buildSearchFilters(parameters);
                    break;
                }
            }

            return filters;
        };

        /**
         * @function getInProgressTasks
         * @description function that get the in progress tasks for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {string} options.searchId - The searchId for the DataTable Suitelet
         * @param {string} options.type - The type for the DataTable Suitelet
         * @returns {Array} inProgress - The inProgress for the DataTable Suitelet
         *
         *
         */

        const getInProgressTasks = (options) => {
            const TITLE = `${MODULE}.GetInProgressTasks`;
            let { searchId, type } = options;

            let taskSearch = SS_Search.load({ id: searchId });
            if (!taskSearch) {
                log.error({ title: TITLE, details: `Unable to load Customer Statement Task saved search.` });
                return;
            }
            taskSearch.Expression = [
                ...taskSearch.Expression,
                'AND',
                [ SS_Constants.CustomRecords.DataTableTask.Fields.TYPE, 'anyof', type ]
            ];

            let inProgress = [];
            let tasks = taskSearch.getResults();
            for (let i = 0, count = tasks.length; i < count; i++) {
                log.debug({ title: `${TITLE} i=${i} task`, details: JSON.stringify(tasks[i]) });
                let contents = tasks[i].data;
                let jsonContents = JSON.parse(contents);
                log.debug({ title: `${TITLE} i=${i} inProgress`, details: JSON.stringify(jsonContents) });

                inProgress = inProgress.concat(jsonContents.list.map(d => d.id));
            }
            inProgress = [ ...new Set(inProgress) ];
            log.debug({ title: `${TITLE} inProgress = ${inProgress.length}`, details: JSON.stringify(inProgress) });

            return inProgress;
        };


        /**
         * @function getTableData
         * @description function that get the table data for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.query - The query for the DataTable Suitelet
         * @param {string} options.searchId - The searchId for the DataTable Suitelet
         * @returns {Object} data - The data for the DataTable Suitelet
         *
         */
        const getTableData = (options) => {
            const TITLE = `${MODULE}.GetTableData`;
            let { query, searchId } = options;
            log.debug({ title: `${TITLE} options`, details: JSON.stringify(options) });

            let dataRowSearch = SS_Search.load({ id: searchId });
            if (!dataRowSearch) {
                log.error({ title: TITLE, details: `Unable to load DataTable rows saved search.` });
                return null;
            }

            log.debug({ title: `${TITLE} parameters`, details: JSON.stringify(query) });
            log.debug({ title: `${TITLE} dataRowSearch.Expression`, details: JSON.stringify(dataRowSearch.Expression) });
            let filters = buildSearchFilters({ parameters: query });
            if (filters.length > 0) {
                filters.unshift('AND');
            }
            // log.debug({ title: `${TITLE} ...dataRowSearch.Expression`, details: JSON.stringify(...dataRowSearch.Expression) });
            log.debug({ title: `${TITLE} filters`, details: JSON.stringify(filters) });

            dataRowSearch.Expression = [ ...dataRowSearch.Expression, ...filters ];
            log.debug({ title: `${TITLE} filterExpression`, details: JSON.stringify(dataRowSearch.Expression) });

            let data = { columns: [], rows: [] };
            let dataRowColumns = dataRowSearch.Columns;
            for (let i = 0, count = dataRowColumns.length; i < count; i++) {
                let label = dataRowColumns[i].label;
                let labelId = SS_String.normalize(label);
                if (labelId.indexOf('_text') > 0) {
                    log.audit({ title: `${TITLE} labelId BEFORE TEXT`, details: `labelId BEFORE TRIM = ${labelId}` });
                    labelId = labelId.substring(0, labelId.indexOf('_text'));
                    log.audit({ title: `${TITLE} labelId AFTER TEXT`, details: `labelId AFTER TRIM = ${labelId}` });
                }
                data.columns.push({
                    data: labelId,
                    title: label
                });
            }
            log.debug({ title: `${TITLE} data.columns`, details: JSON.stringify(data.columns) });

            data.rows = dataRowSearch.getResults();
            return data;
        };

        /**
         * @function process
         * @description function that process the DataTable backend Suitelet for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.request - The request for the DataTable Suitelet
         * @param {Object} options.response - The response for the DataTable Suitelet
         * @returns {Object} data - The data for the DataTable Suitelet
         *
         */
        const process = (options) => {
            const TITLE = `${MODULE}.Process`;
            let { request, response } = options;

            const scriptParams = readScriptParameters({ response });
            if (!scriptParams) { return; }
            log.debug({ title: `${TITLE} scriptParams`, details: JSON.stringify(scriptParams) });

            const queryParameters = readQueryParameters(options);
            if (!queryParameters) { return; }

            log.audit({ title: `${TITLE} queryParameters 2`, details: JSON.stringify(queryParameters) } );

            log.audit('request body', JSON.stringify(request.body));
            log.audit('response process', JSON.stringify(response) );
            if (queryParameters.action.toLowerCase() === 'search') {
                search({
                    body: request.body,
                    query: queryParameters,
                    response,
                    script: scriptParams
                })
            }
            else {
                const handler = HANDLERS[queryParameters.action];
                if (!handler) {
                    sendError({
                        error: `Missing handler for action (${queryParameters.action}).`,
                        response,
                        title: TITLE
                    });
                    return;
                }

                HANDLERS[queryParameters.action].apply(
                    null,
                    [{
                        body: request.body,
                        query: queryParameters,
                        response,
                        script: scriptParams
                    }]
                );
            }
        };

        /**
         * @function readQueryParameters
         * @description function that read the query parameters for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.request - The request for the DataTable Suitelet
         * @param {Object} options.response - The response for the DataTable Suitelet
         * @returns {Object} returnValues - The returnValues for the DataTable Suitelet
         *
         *
         */
        const readQueryParameters = (options) => {
            const TITLE = `${MODULE}.ReadQueryParameters`;                                                                                                                                                                                                                                                  let { request, response } = options;
            log.audit({ title: `${TITLE} request param query read`, details: JSON.stringify(request) });

            let { parameters } = request;
            log.audit({ title: `${TITLE} request param query read`, details: JSON.stringify(parameters) });
            let err = '';

            if (!parameters.action) {
                sendError({
                    error: `Missing required parameter: Action.`,
                    response,
                    title: TITLE
                });
                return null;
            }

            if (!parameters.sltype) {
                sendError({
                    error: `Missing required parameter: Suitelet Type.`,
                    response,
                    title: TITLE
                });
                return null;
            }

            let returnValues = {
                action: parameters.action,
                type: parameters.sltype
            };
            for (const [key, val] of Object.entries(parameters)) {
                returnValues[key] = val;
            }

            return returnValues;
        };

        /**
         * @function readScriptParameters
         * @description function that read the script parameters for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.response - The response for the DataTable Suitelet
         * @returns {Object} scriptParams - The scriptParams for the DataTable Suitelet
         *
         *
         */
        const readScriptParameters = (options) => {
            const TITLE = `${MODULE}.ReadScriptParameters`;
            const PARAMS = SS_Constants.ScriptParameters.DataTableSuiteletBackend;
            let { response } = options;

            let scriptParams = {
                inProgressTaskSearch    : SS_Script.getParameter({ name: PARAMS.InProgressTasks }),
                rowSearch               : SS_Script.getParameter({ name: PARAMS.RowSearch }),
                taskFolder              : SS_Script.getParameter({ name: PARAMS.TaskFolder })
            };

            if (!scriptParams.inProgressTaskSearch) {
                sendError({
                    error: `Missing required value: In Progress Tasks saved search.`,
                    response,
                    title: TITLE
                });
                return null;
            }

            if (!scriptParams.rowSearch) {
                sendError({
                    error: `Missing required value: Data Table Rows saved search.`,
                    response,
                    title: TITLE
                });
                return null;
            }

            if (!scriptParams.taskFolder) {
                sendError({
                    error: `Missing required value: Tasks Folder ID.`,
                    response,
                    title: TITLE
                });
                return null;
            }

                return scriptParams;
        };

        /**
         * @function search
         * @description function that search for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.body - The body for the DataTable Suitelet
         * @param {Object} options.query - The query for the DataTable Suitelet
         * @param {Object} options.response - The response for the DataTable Suitelet
         * @param {Object} options.script - The script for the DataTable Suitelet
         * @returns {Object} data - The data for the DataTable Suitelet
         *
         */
        const search = (options) => {
            const TITLE = `${MODULE}.Search`;
            log.debug({ title: TITLE, details: JSON.stringify(options) });
            let { query, response, script } = options;

            let inProgressTasks = getInProgressTasks({
                searchId: script.inProgressTaskSearch,
                type: query.type
            });

            let tableData = getTableData({
                query,
                searchId: script.rowSearch
            });
            tableData.rows = tableData.rows.filter(row => inProgressTasks.indexOf(row.id) < 0);
            log.debug({ title: `${TITLE} tableData = ${tableData.rows.length}`, details: JSON.stringify([tableData.rows[0]]) });

            response.write({
                output: JSON.stringify({
                    status: true,
                    output: tableData
                })
            });
        };


        /**
         * @function sendError
         * @description function that send error for the DataTable Suitelet
         * @param {Object} options - The options object for the DataTable Suitelet
         * @param {Object} options.error - The error for the DataTable Suitelet
         * @param {Object} options.response - The response for the DataTable Suitelet
         * @param {Object} options.title - The title for the DataTable Suitelet
         * @returns {Object} data - The data for the DataTable Suitelet
         *
         */
        const sendError = (options) => {
            const TITLE = `${MODULE}.SendError`;
            let { error, response, title } = options;

            log.error({ title: `${TITLE}:${title}`, details: JSON.stringify(error) });
            response.write({
                output: JSON.stringify({
                    status: false,
                    error
                })
            });
        };

        return {
            process
        };
    }
);
