/**
 * @NApiVersion         2.1
 * @NModuleScope        SameAccount
 *
 * @fileoverview       Dunning Statements module for creating and sending dunning statements
 * @module             SS_DunningStatements
 * @public
 *
 * @requires           SS_Constants
 * @requires           SS_DataTableTask
 * @requires           SS_Search
 * @requires           SS_Task
 * @requires           SS_UI
 *
 * @exports            buildSearchFilters
 * @exports            render
 * @exports            send
 *
 */

define(
    [
        './SS_Constants',
        './SS_DataTableTask',
        './SS_Search',
        './SS_Task',
        './SS_UI'
    ],
    (
        SS_Constants,
        SS_DataTableTask,
        SS_Search,
        SS_Task,
        SS_UI
    ) => {

        /**
         * @name MODULE
         * @description Module name for logging
         * @type {string}
         */
        const MODULE = `SS|DunningStatements`;

        /**
         * @name FORM
         * @description Form configuration object
         * @type {Object}
         */
        const FORM = SS_Constants.Forms;

        /**
         * @name REQUEST_PARAMS
         * @description Request parameters object
         * @type {Object}
         */
        const REQUEST_PARAMS = SS_Constants.RequestParameters;

        /**
         * @name REQUEST_FIELDS
         * @description Request fields object
         * @type {Object}
         */
        const REQUEST_FIELDS = REQUEST_PARAMS;

        /**
         * @name OPERATOR
         * @description Search operator object
         * @type {Object}
         */
        const OPERATOR = SS_Search.Operator;

        /**
         * @name TRANSACTION
         * @description Transaction object
         * @type {Object}
         */
        const TRANSACTION = SS_Constants.Transaction;

        /**
         * @name REQUEST_TYPE
         * @description Request type object
         * @type {{DUNNING_LETTERS: string, DUNNING_STATEMENTS: string}}
         */
        const REQUEST_TYPE = SS_Constants.CustomLists.DataTableSuiteletTypes

        /**
         * @function buildSearchFilters
         * @description Build search filters for dunning statements
         * @param {Object} options
         * @param {string} options.balance
         * @param {string} options.cas
         * @param {string} options.condition
         * @param {string} options.customer
         * @param {string} options.lastcomm
         * @param {string} options.levels
         * @param {string} options.openbalanceusd
         * @param {string} options.subsidiary
         * @returns {Array}
         */
        const buildSearchFilters = (options) => {
            const TITLE = `${MODULE}.BuildSearchFilters`;
            let {
                // balance,
                cas,
                condition,
                customer,
                lastcomm,
                levels,
                openbalanceusd,
                subsidiary
            } = options;
            log.debug({ title: `${TITLE} options`, details: JSON.stringify(options) });
            let filters = [];

            if (cas) {
                filters = filters.concat([
                    [ `customermain.${SS_Constants.Entity.CustomFields.CAS}`, OPERATOR.ANYOF, cas ],
                    'AND'
                ]);
            }

            if (customer) {
                filters = filters.concat([
                    [ TRANSACTION.Fields.NAME, OPERATOR.ANYOF, customer ],
                    'AND'
                ]);
            }

            if (lastcomm) {
                filters = filters.concat([
                    [
                        [ TRANSACTION.CustomFields.LAST_COMMUNICATION_DATE, OPERATOR.ONORBEFORE, lastcomm ],
                        'OR',
                        [ TRANSACTION.CustomFields.LAST_COMMUNICATION_DATE, OPERATOR.ISEMPTY, null ]
                    ],
                    'AND'
                ]);
            }

            if (subsidiary) {
                filters = filters.concat([
                    [
                        TRANSACTION.Fields.SUBSIDIARY, OPERATOR.ANYOF, subsidiary
                    ],
                    'AND'
                ]
                );
            }

            if (condition && !!openbalanceusd === true) {
                let operator = OPERATOR.EQUALTO;
                switch (condition.toLowerCase()) {
                    case 'equalgreat': {
                        operator = OPERATOR.GREATERTHANOREQUALTO;
                        break;
                    }
                    case 'equalless': {
                        operator = OPERATOR.LESSTHANOREQUALTO;
                        break;
                    }
                }
                filters = filters.concat([
                    [ `sum(${TRANSACTION.Fields.AMOUNT_REMAINING})`, operator, openbalanceusd ],
                    'AND'
                ]);
            }

            let dunningLevel = SS_Constants.CustomLists.DunningLevels.Values.find(d => d.id === levels);
            if (dunningLevel) {
                filters = filters.concat([
                    [ `max(${TRANSACTION.Fields.DAYS_OVERDUE})`, OPERATOR.BETWEEN, dunningLevel.from, dunningLevel.to ],
                    'AND'
                ]);
            }

            if (filters.length > 0) {
                filters.pop();
            }
            log.debug({ title: `${TITLE} END`, details: JSON.stringify(filters) });

            return filters;
        };

        /**
         * @function send
         * @description Send dunning statements
         * @param {Object} options
         * @param {Object} options.body
         * @param {Object} options.query
         * @param {Object} options.response
         * @param {Object} options.script
         * @returns {void}
         */
        const send = (options) => {
            const TITLE = `${MODULE}.Send`;
            let { body, query, response, script } = options;
            log.debug({ title: `${TITLE} body`, details: body });

            // Build customers array
            /* let jsonBody = JSON.parse(body);
            let customers = jsonBody.list.map(d => {
                let split = d.split('-');
                return {
                    customer: split[0],
                    level: 'Statement',
                    overrideEmails: d.override,
                    subsidiary: split[1]
                };
            });
            log.debug({ title: `${TITLE} customers`, details: JSON.stringify(customers) }); */

            // Create MR Task
            let dtTask = SS_DataTableTask.create({
                data: body,
                type: SS_Constants.CustomLists.DataTableSuiteletTypes.DUNNING_STATEMENTS
            });
            log.debug({ title: `${TITLE} dtTask`, details: JSON.stringify(dtTask) });

            let mrTaskParams = {};
            mrTaskParams[SS_Constants.ScriptParameters.DunningStatementMapReduce.TaskRecord] = dtTask.data;
            // mrTaskParams[SS_Constants.ScriptParameters.DataTableTaskMapReduce.CustomersArray] = customers;
            SS_Task.createMapReduceTask({
                scriptId: SS_Constants.Scripts.DunningStatementsMapReduce.scriptId,
                params: mrTaskParams
            });

            // log.debug({ title: TITLE, details: `Successfully created map/reduce task ID ${mrTask.data}.` });
            response.write({
                output: JSON.stringify(dtTask)
            });
        };

        /**
         * @function render
         * @description Render the dunning statements form
         * @param {Object} options
         * @param {Object} options.request
         * @param {Object} options.response
         * @param {Object} options.defaultValues
         * @returns {Object}
         */
        const render = (options) => {
            const TITLE = `${MODULE}.Render`;
            let { request, response, defaultValues } = options;
            log.audit({ title: `${TITLE} request`, details: JSON.stringify(request) });
            log.audit({ title: `${TITLE} defaultValues`, details: JSON.stringify(defaultValues) });
            log.audit({ title: `${TITLE} defaultValues.custpage_dunning_type`, details: defaultValues.custpage_dunning_type })
            switch (defaultValues.custpage_dunning_type) {
                case REQUEST_TYPE.DUNNING_STATEMENTS: {
                    let queryParameters = SS_UI.readQueryParameters({ fields: REQUEST_FIELDS.DUNNING_STATEMENTS, request });
                    log.debug({ title: `${TITLE} queryParameters`, details: JSON.stringify(queryParameters) });
                    let parameterFields = SS_UI.mapParametersToFields({
                        fields: REQUEST_FIELDS.DUNNING_STATEMENTS,
                        form: FORM.DUNNING_STATEMENTS,
                        parameters: queryParameters
                    });
                    log.debug({ title: `${TITLE} parameterFields`, details: JSON.stringify(parameterFields) });
                    defaultValues = {
                        ...defaultValues,
                        ...parameterFields
                    };
                    break;
                }

                case REQUEST_TYPE.DUNNING_LETTERS: {
                    let queryParameters = SS_UI.readQueryParameters({ fields: REQUEST_FIELDS.DUNNING_LETTERS, request });
                    log.debug({ title: `${TITLE} queryParameters`, details: JSON.stringify(queryParameters) });
                    let parameterFields = SS_UI.mapParametersToFields({
                        fields: REQUEST_FIELDS.DUNNING_LETTERS,
                        form: FORM.DUNNING_LETTERS,
                        parameters: queryParameters
                    });
                    log.debug({ title: `${TITLE} parameterFields`, details: JSON.stringify(parameterFields) });
                    defaultValues = {
                        ...defaultValues,
                        ...parameterFields
                    };
                    break;
                }

            }
       //     let queryParameters = SS_UI.readQueryParameters({ fields: REQUEST_FIELDS.DUNNING_STATEMENTS, request });

          //  log.debug({ title: `${TITLE} queryParameters`, details: JSON.stringify(queryParameters) });

            // let parameterFields = SS_UI.mapParametersToFields({
            //     fields: REQUEST_FIELDS.DUNNING_STATEMENTS,
            //     form: FORM,
            //     parameters: queryParameters
            // });
       //     log.debug({ title: `${TITLE} parameterFields`, details: JSON.stringify(parameterFields) });

            // defaultValues = {
            //     ...defaultValues,
            //     ...parameterFields
            // };
            log.debug({ title: `${TITLE} defaultValues`, details: JSON.stringify(defaultValues) });
            switch (defaultValues.custpage_dunning_type) {
                case REQUEST_TYPE.DUNNING_STATEMENTS: {
                    return SS_UI.buildForm({
                        config: FORM.DUNNING_STATEMENTS,
                        defaultValues: defaultValues
                    });
                    break;
                }

                case REQUEST_TYPE.DUNNING_LETTERS: {
                    return SS_UI.buildForm({
                        config: FORM.DUNNING_LETTERS,
                        defaultValues: defaultValues
                    });
                    break;
                }

            }
            // return SS_UI.buildForm({
            //     config: FORM,
            //     defaultValues: defaultValues
            // });
        };

        return {
            buildSearchFilters,
            render,
            send
        };
    }
);
