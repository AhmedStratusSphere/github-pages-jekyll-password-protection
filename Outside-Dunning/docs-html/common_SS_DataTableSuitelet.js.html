        /**
         * @function getDefaultUrlValues
         * @description Retrieves the default URL values for the Suitelet
         * @param {Object} options - The options object for the Suitelet
         * @param {int} options.type - The type of Suitelet
         * @returns {Object} The default URL values for the Suitelet
         *
         * @example
         * getDefaultUrlValues({ type: 101 });
         */
        const getDefaultUrlValues = (options) => {
            const TITLE = `${MODULE}.GetDefaultUrlValues`;
            const SCRIPT = SS_Constants.Scripts.DataTableSuitelet;
            log.debug({ title: `${TITLE} SCRIPT`, details: JSON.stringify(SCRIPT) });
            let deploymentId = null;

            log.audit('SCRIPT 1', JSON.stringify(SCRIPT));
            log.audit('option 1', JSON.stringify(options));
            switch (options.type) {
                case TYPES.DUNNING_STATEMENTS: {
                    deploymentId = SCRIPT.DUNNING_STATEMENTS.deploymentId;
                    break;
                }
                case TYPES.DUNNING_LETTERS: {
                    deploymentId = SCRIPT.DUNNING_LETTERS.deploymentId; //TODO: Change this to the correct deployment ID.
                    break;
                }
                //Add letter case for  Suitelet types here.
            }


            log.audit('SCRIPT 2', JSON.stringify(SCRIPT));
            log.audit('option 2', JSON.stringify(options));
            return {
                Backend: SS_Url.resolveScript({
                    scriptId: SCRIPT.BACKEND.scriptId,
                    deploymentId
                }),
                Transaction: '/app/accounting/transactions/transaction.nl?id='
            };
        };

        /**
         * @function getRenderedForm
         * @description Retrieves the rendered form for the Suitelet
         * @param {Object} options - The options object for the Suitelet
         * @param {Object} options.renderOptions - The render options for the Suitelet
         * @param {string} options.type - The type of Suitelet
         * @returns {nlobjForm} The rendered form for the Suitelet
         *
         */
        const getRenderedForm = (options) => {
            const TITLE = `${MODULE}.GetRenderedForm`;
            let { renderOptions, type } = options;
            let renderedForm = null;

            switch (options.type) {
                case TYPES.DUNNING_STATEMENTS: {
                    renderedForm = SS_DunningStatements.render(renderOptions);
                    break;
                }

                // TODO: Cases can be added here for additional Suitelet types.

                case TYPES.DUNNING_LETTERS: {
                    renderedForm = SS_DunningStatements.render(renderOptions);
                    break;
                }

            }

            log.audit('option type 3 ', JSON.stringify(options.type));

            log.audit({ title: 'rendred form 2', details: `renderedForm = ${renderedForm}` });
            return renderedForm;
        };

        /**
         * @function readScriptParameters
         * @description Reads the script parameters for the Suitelet
         * @returns {Object} The script parameters for the Suitelet
         * @throws {error.SuiteScriptError} MISSING_REQD_ARGUMENT if a required parameter is missing
         */
        const readScriptParameters = () => {
            const TITLE = `${MODULE}.ReadScriptParameters`;
            const PARAMS = SS_Constants.ScriptParameters.DataTableSuitelet;

            let scriptParams = {
                type: SS_Script.getParameter({ name: PARAMS.Type })
            };
            log.audit('PARAMS', JSON.stringify(scriptParams));
            if (!scriptParams.type) {
                log.error({ title: TITLE, details: `Missing required value: Type.` });
                return null;
            }

            return scriptParams;
        };

        /**
         * @function render
         * @description Renders the Suitelet
         * @param {Object} options - The options object for the Suitelet
         * @param {Object} options.request - The request object for the Suitelet
         * @param {Object} options.response - The response object for the Suitelet
         * @returns {nlobjForm} The rendered form for the Suitelet
         *
         * @example
         * render({
         *    request: request,
         *    response: response
         *    });
         */
        const render = (options) => {
            const TITLE = `${MODULE}.Render`;

            const scriptParams = readScriptParameters();  //object with type  script param {type : 101 }
            if (!scriptParams) { return; }
            log.debug({ title: `${TITLE} scriptParams`, details: JSON.stringify(scriptParams) });

            let { request, response } = options;

            log.audit('request : ', JSON.stringify(request));
            log.audit('response : ', JSON.stringify(response));

            let renderOptions = {
                request,
                response,
                defaultValues: {
                    custpage_url_values: JSON.stringify(getDefaultUrlValues({ type: scriptParams.type })) , //101
                    custpage_dunning_type: scriptParams.type
                    /* custpage_table_css: `<style>body {
                        transition: background-color 2s cubic-bezier(1, 1, 1, 1);
                        transition-delay: 0s;
                    }</style>` */
                }
            };

            log.audit('scriptPrama default url value ', JSON.stringify(getDefaultUrlValues({ type: scriptParams.type })))

            log.audit('renderOptions', JSON.stringify(renderOptions));
            log.audit('type', scriptParams.type);

            let renderedForm = getRenderedForm({
                type: scriptParams.type,
                renderOptions
            });
            log.audit('renderedForm', JSON.stringify(renderedForm));
            if (!renderedForm) {
                log.error({ title: TITLE, details: `Unable to render form. Please see Execution Logs for details.` });
                return;
            }


            writeDataTable({
                form: renderedForm,
                template: SS_Constants.Templates.DataTableSuitelet
            });
            response.writePage({ pageObject: renderedForm });
            log.debug({ title: TITLE, details: `Successfully built Suitelet form.` });

        };

        /**
         * @function writeDataTable
         * @description Writes the data table to the Suitelet form
         * @param {Object} options - The options object for the data table
         * @param {nlobjForm} options.form - The form object for the Suitelet
         * @param {string} options.template - The template for the data table
         * @retyurns {void}
         */
        const writeDataTable = (options) => {
            let title = `${MODULE}.WriteDataTable`;
            let { form, template } = options;
            let tableContents = SS_File.read({ name: template });
            log.debug({ title: title, details: tableContents });

            let tableField = SS_UI.getField({ id: 'custpage_table', parent: form });
            tableField.defaultValue = tableContents;
        };

        return {
            render
        };
    }
);
